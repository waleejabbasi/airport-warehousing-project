type='csv',
field_delimiter=',',
skip_header=1,
field_optionally_enclosed_by='"'


--CREATE STAGE
CREATE OR REPLACE STAGE AIRPORT_STAGE
file_format=myformat,
storage_integration=airport_integration,
url='s3://airport-raw-datawal/'

--CREATE SNOWPIPE

auto_ingest=TRUE AS
COPY INTO raw_data_table
FROM @AIRPORT_STAGE
FILE_FORMAT=(FORMAT_NAME=myformat)
